# 机器翻译

## 一句话定义

机器翻译（Machine Translation, MT）是利用计算机将一种自然语言（源语言）自动转换为另一种自然语言（目标语言）的技术。

## 核心公式

### 1. 翻译概率（统计机器翻译）

$$P(e|f) = \frac{P(e) \cdot P(f|e)}{P(f)} \propto P(e) \cdot P(f|e)$$

### 2. Seq2Seq 条件概率

$$P(y|x) = \prod_{t=1}^{T_y} P(y_t|y_1, ..., y_{t-1}, x)$$

### 3. Attention 权重

$$\alpha_{ij} = \frac{\exp(e_{ij})}{\sum_{k=1}^{T_x} \exp(e_{ik})}$$

其中 $e_{ij} = \text{score}(h_i, s_j)$ 表示编码器第 i 个位置与解码器第 j 个位置的关联程度。

### 4. Transformer 缩放点积注意力

$$\text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^T}{\sqrt{d_k}}\right)V$$

### 2. 统计机器翻译

#### 2.1 IBM 模型

**核心思想：** 通过词对齐和概率模型学习翻译规则。

```python
# IBM Model 1 简化实现
import numpy as np
from collections import defaultdict

def ibm_model1(source_corpus, target_corpus, num_iterations=10):
    """
    IBM Model 1: 词对齐翻译模型
    source_corpus: 源语言句子列表
    target_corpus: 目标语言句子列表
    """
    vocab_src = set(' '.join(source_corpus).split())
    vocab_tgt = set(' '.join(target_corpus).split())

    # 初始化翻译概率均匀分布
    t = defaultdict(lambda: 1.0 / len(vocab_src))

    for iteration in range(num_iterations):
        count = defaultdict(float)
        total = defaultdict(float)

        for src_sent, tgt_sent in zip(source_corpus, target_corpus):
            src_words = src_sent.split()
            tgt_words = tgt_sent.split()

            # E-step: 计算每个目标词由源词翻译的概率
            for tgt_word in tgt_words:
                total[tgt_word] = 0
                for src_word in src_words:
                    total[tgt_word] += t[(tgt_word, src_word)]

            # M-step: 更新翻译概率
            for tgt_word in tgt_words:
                for src_word in src_words:
                    count[(tgt_word, src_word)] += t[(tgt_word, src_word)] / total[tgt_word]
                    count[src_word] += t[(tgt_word, src_word)] / total[tgt_word]

        # 更新翻译概率
        for (tgt, src), cnt in count.items():
            t[(tgt, src)] = cnt / count.get(src, 1e-10)

    return t
```

### 3. Seq2Seq 模型

#### 3.1 编码器-解码器架构

**编码器：** 将源序列编码为固定维度向量
```python
import torch
import torch.nn as nn

class Encoder(nn.Module):
    def __init__(self, vocab_size, embed_dim, hidden_dim):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        self.rnn = nn.GRU(embed_dim, hidden_dim, batch_first=True)

    def forward(self, x):
        embedded = self.embedding(x)
        outputs, hidden = self.rnn(embedded)
        return outputs, hidden
```

**解码器：** 逐词生成目标序列
```python
class Decoder(nn.Module):
    def __init__(self, vocab_size, embed_dim, hidden_dim):
        super().__init__()
        self.embedding = nn.Embedding(vocab_size, embed_dim)
        self.rnn = nn.GRU(embed_dim, hidden_dim, batch_first=True)
        self.fc = nn.Linear(hidden_dim, vocab_size)

    def forward(self, x, hidden):
        embedded = self.embedding(x.unsqueeze(1))
        output, hidden = self.rnn(embedded, hidden)
        prediction = self.fc(output.squeeze(1))
        return prediction, hidden
```

**Seq2Seq 完整模型：**
```python
class Seq2Seq(nn.Module):
    def __init__(self, encoder, decoder):
        super().__init__()
        self.encoder = encoder
        self.decoder = decoder

    def forward(self, src, tgt, teacher_forcing_ratio=0.5):
        batch_size = src.size(0)
        tgt_len = tgt.size(1)
        tgt_vocab_size = self.decoder.fc.out_features

        outputs = torch.zeros(batch_size, tgt_len, tgt_vocab_size)

        _, hidden = self.encoder(src)

        # 解码器初始输入为 <BOS>
        decoder_input = tgt[:, 0]

        for t in range(1, tgt_len):
            output, hidden = self.decoder(decoder_input, hidden)
            outputs[:, t] = output

            # Teacher Forcing
            teacher_force = torch.rand(1).item() < teacher_forcing_ratio
            top1 = output.argmax(1)

            decoder_input = tgt[:, t] if teacher_force else top1

        return outputs
```

### 4. Attention 机制

#### 4.1 为什么需要 Attention

**问题：** 固定维度向量成为信息瓶颈，无法有效处理长序列。

**解决：** 允许解码器在生成每个词时关注源序列的不同部分。

```python
class Attention(nn.Module):
    def __init__(self, hidden_dim):
        super().__init__()
        self.attn = nn.Linear(hidden_dim * 2, hidden_dim)
        self.v = nn.Linear(hidden_dim, 1, bias=False)

    def forward(self, encoder_outputs, decoder_hidden):
        """
        encoder_outputs: (batch, src_len, hidden_dim)
        decoder_hidden: (batch, hidden_dim)
        """
        src_len = encoder_outputs.size(1)

        # 重复decoder_hidden以匹配encoder_outputs长度
        decoder_hidden = decoder_hidden.unsqueeze(1).repeat(1, src_len, 1)

        # 计算注意力能量
        energy = torch.tanh(self.attn(torch.cat((decoder_hidden, encoder_outputs), dim=2)))
        attention = self.v(energy).squeeze(2)

        # Softmax归一化
        attention_weights = torch.softmax(attention, dim=1)

        return attention_weights
```

### 5. Transformer 架构

#### 5.1 整体架构

```
输入 -> 嵌入 + 位置编码 -> 编码器栈(Nx) -> 解码器栈(Nx) -> 输出层
```

#### 5.2 多头注意力实现

```python
class MultiHeadAttention(nn.Module):
    def __init__(self, embed_dim, num_heads):
        super().__init__()
        assert embed_dim % num_heads == 0

        self.embed_dim = embed_dim
        self.num_heads = num_heads
        self.head_dim = embed_dim // num_heads

        self.qkv_proj = nn.Linear(embed_dim, embed_dim * 3)
        self.out_proj = nn.Linear(embed_dim, embed_dim)

    def forward(self, x, mask=None):
        B, T, C = x.shape

        # 生成 Q, K, V 并分头
        qkv = self.qkv_proj(x).reshape(B, T, 3, self.num_heads, self.head_dim)
        q, k, v = qkv.unbind(2)

        # 缩放点积注意力
        scale = math.sqrt(self.head_dim)
        scores = torch.matmul(q, k.transpose(-2, -1)) / scale

        if mask is not None:
            scores = scores.masked_fill(mask == 0, float('-inf'))

        attn = torch.softmax(scores, dim=-1)
        out = torch.matmul(attn, v)

        return self.out_proj(out.reshape(B, T, C))
```

#### 5.3 Transformer 编码器

```python
class TransformerEncoderLayer(nn.Module):
    def __init__(self, embed_dim, num_heads, ff_dim, dropout=0.1):
        super().__init__()
        self.self_attn = MultiHeadAttention(embed_dim, num_heads)
        self.ff = nn.Sequential(
            nn.Linear(embed_dim, ff_dim),
            nn.ReLU(),
            nn.Linear(ff_dim, embed_dim)
        )
        self.norm1 = nn.LayerNorm(embed_dim)
        self.norm2 = nn.LayerNorm(embed_dim)
        self.dropout = nn.Dropout(dropout)

    def forward(self, x, mask=None):
        # 自注意力 + 残差
        attn_out = self.self_attn(x, mask)
        x = self.norm1(x + self.dropout(attn_out))

        # 前馈网络 + 残差
        ff_out = self.ff(x)
        x = self.norm2(x + self.dropout(ff_out))

        return x
```

#### 5.4 完整 Transformer 翻译模型

```python
class TransformerMT(nn.Module):
    def __init__(self, vocab_src, vocab_tgt, embed_dim=512, num_heads=8,
                 num_layers=6, ff_dim=2048, dropout=0.1):
        super().__init__()

        self.encoder = nn.ModuleList([
            TransformerEncoderLayer(embed_dim, num_heads, ff_dim, dropout)
            for _ in range(num_layers)
        ])

        self.decoder = nn.ModuleList([
            TransformerDecoderLayer(embed_dim, num_heads, ff_dim, dropout)
            for _ in range(num_layers)
        ])

        self.src_embed = nn.Embedding(vocab_src, embed_dim)
        self.tgt_embed = nn.Embedding(vocab_tgt, embed_dim)

        self.pos_encoding = PositionalEncoding(embed_dim, dropout)
        self.fc_out = nn.Linear(embed_dim, vocab_tgt)

    def forward(self, src, tgt):
        # 嵌入 + 位置编码
        src = self.pos_encoding(self.src_embed(src))
        tgt = self.pos_encoding(self.tgt_embed(tgt))

        # 编码
        enc_out = src
        for layer in self.encoder:
            enc_out = layer(enc_out)

        # 解码
        dec_out = tgt
        for layer in self.decoder:
            dec_out = layer(dec_out, enc_out)

        return self.fc_out(dec_out)
```

# 使用Hugging Face Transformers进行翻译
from transformers import AutoModelForSeq2SeqLM, AutoTokenizer

model_name = "Helsinki-NLP/opus-mt-en-zh"
tokenizer = AutoTokenizer.from_pretrained(model_name)
model = AutoModelForSeq2SeqLM.from_pretrained(model_name)

def translate(text):
    inputs = tokenizer(text, return_tensors="pt", padding=True)
    outputs = model.generate(**inputs)
    return tokenizer.decode(outputs[0], skip_special_tokens=True)

# 示例
english_text = "Machine translation is a fascinating field of artificial intelligence."
