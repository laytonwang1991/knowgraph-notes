# 模拟学习（Sim-to-Real）

## 一句话定义

模拟学习（Sim-to-Real）是将机器人在仿真环境中训练的技能迁移到真实世界的过程，通过域随机化、系统识别等技术弥合仿真与现实之间的领域差距。

## 详细说明

### 1. 为什么需要 Sim-to-Real？

在真实机器人上直接训练存在以下问题：

| 问题 | 描述 | 仿真优势 |
|------|------|----------|
| **样本效率** | 真实机器人数据采集慢、成本高 | 并行仿真可加速 1000x |
| **安全性** | 训练失败可能损坏机器人或环境 | 零物理风险 |
| **可复现性** | 真实环境难以精确复现 | 仿真完全可复现 |
| **边界探索** | 危险动作在现实中无法测试 | 可自由探索边界 |

### 2. Sim-to-Real 挑战的核心问题

Sim-to-Real 迁移面临的主要挑战是**领域差异（Domain Gap）**：

```
仿真环境                    真实环境
─────────────────────────────────────────────
物理参数精确已知    ≠    物理参数存在不确定性
传感器完美无噪      ≠    传感器存在噪声和延迟
视觉渲染逼真但失真  ≠    真实视觉纹理复杂
关节控制精确        ≠    控制存在摩擦、死区
──────────────────    ────────────────────
这些差异导致仿真训练的策略无法直接泛化到真实环境
```

### 3. 主要方法

#### 3.1 域随机化（Domain Randomization）

通过在仿真中随机化物理参数、视觉渲染等，使策略学习到域不变特征。

**域随机化的随机化维度**：

```python
class DomainRandomizer:
    """域随机化：随机化仿真参数以提高泛化能力"""

    def __init__(self):
        # 物理参数范围
        self.physics_ranges = {
            'gravity': (9.7, 9.81),           # 重力 (m/s²)
            'link_mass': (0.8, 1.2),           # 连杆质量倍数
            'friction': (0.3, 1.5),            # 摩擦系数
            'motor_strength': (0.85, 1.15),   # 电机强度
            'joint_damping': (0.5, 2.0),       # 关节阻尼
            'control_latency': (0.001, 0.02), # 控制延迟 (s)
        }

        # 视觉参数范围
        self.visual_ranges = {
            'light_intensity': (0.5, 1.5),
            'light_position': (-2, 2, 0.5, 2, -2, 2),
            'camera_noise': (0, 0.01),
            'texture_variation': (0.8, 1.2),
            'background_color': ((0, 0, 0), (255, 255, 255)),
        }

    def randomize_physics(self, env):
        """随机化物理参数"""
        for param_name, (low, high) in self.physics_ranges.items():
            if param_name == 'gravity':
                env.set_gravity(np.random.uniform(low, high))
            elif param_name == 'link_mass':
                for link in env.robot.get_links():
                    link.mass *= np.random.uniform(low, high)
            elif param_name == 'friction':
                env.set_friction(np.random.uniform(low, high))
            elif param_name == 'motor_strength':
                env.set_motor_strength(np.random.uniform(low, high))
            elif param_name == 'joint_damping':
                env.set_joint_damping(np.random.uniform(low, high))
            elif param_name == 'control_latency':
                env.set_control_latency(np.random.uniform(low, high))

    def randomize_visual(self, renderer):
        """随机化视觉渲染参数"""
        # 光照强度
        renderer.set_light_intensity(
            np.random.uniform(*self.visual_ranges['light_intensity'])
        )

        # 光照位置
        pos_range = self.visual_ranges['light_position']
        renderer.set_light_position(
            np.random.uniform(pos_range[0], pos_range[1]),
            np.random.uniform(pos_range[2], pos_range[3]),
            np.random.uniform(pos_range[4], pos_range[5])
        )

        # 相机噪声
        renderer.set_camera_noise(
            np.random.uniform(*self.visual_ranges['camera_noise'])
        )

        # 纹理缩放
        renderer.set_texture_scale(
            np.random.uniform(*self.visual_ranges['texture_variation'])
        )

        # 背景颜色
        bg = self.visual_ranges['background_color']
        renderer.set_background_color(
            np.random.uniform(bg[0][0], bg[1][0]) / 255,
            np.random.uniform(bg[0][1], bg[1][1]) / 255,
            np.random.uniform(bg[0][2], bg[1][2]) / 255
        )

    def randomize_all(self, env, renderer=None):
        """随机化所有参数"""
        self.randomize_physics(env)
        if renderer is not None:
            self.randomize_visual(renderer)
```

```python
class VisualDomainRandomization:
    """视觉域随机化：改变视觉外观以提高泛化"""

    def __init__(self):
        self.textures = self.load_random_textures()

    def apply_color_randomization(self, image):
        """颜色随机化"""
        # 色调偏移
        hsv = self.rgb_to_hsv(image)
        hsv[:, :, 0] = (hsv[:, :, 0] + np.random.uniform(-0.1, 0.1)) % 1.0
        image = self.hsv_to_rgb(hsv)

        # 饱和度变化
        image = image * np.random.uniform(0.7, 1.3)
        image = np.clip(image, 0, 1)

        return image

    def apply_texture_randomization(self, obj, obj_id):
        """纹理随机化"""
        new_texture = random.choice(self.textures)
        obj.set_texture(new_texture, obj_id)

    def apply_occlusion(self, image, num_occluders=3):
        """添加遮挡物"""
        h, w = image.shape[:2]

        for _ in range(num_occluders):
            # 随机位置和大小的遮挡矩形
            x = np.random.randint(0, w - 20)
            y = np.random.randint(0, h - 20)
            w_occ = np.random.randint(10, 50)
            h_occ = np.random.randint(10, 50)

            # 随机颜色
            color = np.random.randint(0, 255, 3)

            image[y:y+h_occ, x:x+w_occ] = color

        return image

    def apply_noise(self, image, noise_type='gaussian'):
        """添加传感器噪声"""
        if noise_type == 'gaussian':
            noise = np.random.normal(0, 5, image.shape)
            image = image + noise
        elif noise_type == 'salt_pepper':
            prob = 0.01
            salt = np.random.random(image.shape) < prob / 2
            pepper = np.random.random(image.shape) < prob / 2
            image[salt] = 255
            image[pepper] = 0

        return np.clip(image, 0, 255).astype(np.uint8)
```

#### 3.2 系统识别（System Identification）

精确建模真实物理系统，减少仿真与真实的差距。

```python
class SystemIdentifier:
    """系统识别：估计真实物理参数"""

    def __init__(self, simulation_model):
        self.sim_model = simulation_model
        self.param_history = []

    def collect_trajectory_data(self, robot, num_trajectories=100):
        """收集真实机器人轨迹数据"""
        trajectories = []

        for _ in range(num_trajectories):
            # 随机初始化
            q_init = robot.random_configuration()
            robot.set_state(q_init, np.zeros(robot.dof))

            trajectory = {'states': [], 'actions': [], 'next_states': []}

            for t in range(100):  # 100步轨迹
                state = robot.get_state()

                # 随机动作
                action = np.random.uniform(-1, 1, robot.dof)
                robot.execute_action(action)

                next_state = robot.get_state()

                trajectory['states'].append(state)
                trajectory['actions'].append(action)
                trajectory['next_states'].append(next_state)

            trajectories.append(trajectory)

        return trajectories

    def compute_dynamics_error(self, params, trajectories):
        """计算给定参数下的动力学预测误差"""
        self.sim_model.set_params(params)

        total_error = 0
        total_samples = 0

        for traj in trajectories:
            for state, action, next_state_real in zip(
                traj['states'], traj['actions'], traj['next_states']
            ):
                # 仿真一步预测
                next_state_pred = self.sim_model.step(state, action)

                # 计算误差
                error = np.linalg.norm(next_state_pred - next_state_real)
                total_error += error
                total_samples += 1

        return total_error / total_samples

    def estimate_parameters(self, trajectories, initial_params=None):
        """估计真实物理参数"""
        if initial_params is None:
            initial_params = self.sim_model.get_nominal_params()

        # 使用优化方法估计参数
        def objective(params):
            return self.compute_dynamics_error(params, trajectories)

        # 梯度下降优化
        params = initial_params.copy()
        learning_rate = 0.01

        for iteration in range(1000):
            error = objective(params)

            if iteration % 100 == 0:
                print(f"Iteration {iteration}, Error: {error:.6f}")

            # 简化的梯度估计
            gradient = self.estimate_gradient(objective, params)
            params = params - learning_rate * gradient

            # 参数约束（物理参数应为正）
            params = np.maximum(params, 0.01)

            self.param_history.append(params.copy())

        return params

    def estimate_gradient(self, objective, params, eps=1e-5):
        """估计目标函数的梯度"""
        gradient = np.zeros_like(params)

        for i in range(len(params)):
            params_plus = params.copy()
            params_plus[i] += eps
            params_minus = params.copy()
            params_minus[i] -= eps

            gradient[i] = (objective(params_plus) - objective(params_minus)) / (2 * eps)

        return gradient
```

#### 3.3 烧入效应（Burn-in Effects）

初始仿真阶段的处理策略，确保策略在真实环境中稳定启动。

```python
class BurnInStrategy:
    """烧入策略：处理Sim-to-Real初始阶段"""

    def __init__(self, sim_env, real_env):
        self.sim_env = sim_env
        self.real_env = real_env
        self.burn_in_steps = 1000

    def collect_burn_in_data(self, policy, random_actions=True):
        """
        收集烧入数据：在真实环境初期使用安全动作

        Args:
            policy: 训练好的策略网络
            random_actions: 是否使用随机动作（更保守）
        """
        burn_in_buffer = {
            'states': [],
            'actions': [],
            'rewards': [],
            'next_states': []
        }

        env = self.real_env
        state = env.reset()

        for step in range(self.burn_in_steps):
            if random_actions or step < self.burn_in_steps * 0.3:
                # 初期使用随机或小动作
                action = np.random.uniform(-0.3, 0.3, env.action_dim)
            else:
                # 后期逐渐使用策略动作
                action = policy.select_action(state)

            next_state, reward, done, info = env.step(action)

            burn_in_buffer['states'].append(state)
            burn_in_buffer['actions'].append(action)
            burn_in_buffer['rewards'].append(reward)
            burn_in_buffer['next_states'].append(next_state)

            state = next_state

            if done:
                state = env.reset()

        return burn_in_buffer

    def adaptive_burn_in(self, policy, initial_safe_steps=500):
        """
        自适应烧入：根据实际执行情况调整

        监控策略在真实环境的表现，动态调整探索程度
        """
        state = self.real_env.reset()
        safe_actions = 0
        total_actions = 0

        for step in range(self.burn_in_steps):
            # 渐进式策略使用
            use_policy_prob = min(1.0, step / initial_safe_steps)

            if np.random.random() < use_policy_prob:
                action = policy.select_action(state)
            else:
                action = np.random.uniform(-0.2, 0.2, self.real_env.action_dim)

            next_state, reward, done, info = self.real_env.step(action)

            # 检查动作是否"安全"（小误差范围内）
            if info.get('tracking_error', float('inf')) < 0.05:
                safe_actions += 1

            total_actions += 1
            state = next_state

            if done:
                state = self.real_env.reset()

            # 动态调整：如果安全动作比例低，降低策略使用概率
            if total_actions > 100:
                safe_ratio = safe_actions / total_actions
                if safe_ratio < 0.8:
                    use_policy_prob *= 0.95

        return safe_ratio
```

#### 3.4 真实世界微调（Real-World Fine-tuning）

在真实环境中微调仿真训练的策略。

```python
class RealWorldFineTuner:
    """真实世界微调：在线调整仿真训练策略"""

    def __init__(self, policy, value_network, real_env,
                 exploration_noise=0.1, learning_rate=0.0003):
        self.policy = policy
        self.value_network = value_network
        self.env = real_env
        self.exploration_noise = exploration_noise
        self.lr = learning_rate

        # 经验回放
        self.replay_buffer = ReplayBuffer(capacity=100000)

        # 安全约束
        self.action_bound = 1.0
        self.safe_region = {'lower': -2.0, 'upper': 2.0}

    def add_exploration_noise(self, action):
        """添加探索噪声"""
        noise = np.random.normal(0, self.exploration_noise, size=action.shape)
        action = np.clip(action + noise, -self.action_bound, self.action_bound)
        return action

    def is_safe_state(self, state):
        """检查状态是否在安全范围内"""
        for i, (low, high) in enumerate(
            zip(self.safe_region['lower'], self.safe_region['upper'])
        ):
            if not (low <= state[i] <= high):
                return False
        return True

    def collect_real_data(self, num_episodes=10, max_steps_per_episode=1000):
        """在真实环境中收集数据"""
        for episode in range(num_episodes):
            state = self.env.reset()
            episode_reward = 0

            for step in range(max_steps_per_episode):
                # 选择动作用策略
                with torch.no_grad():
                    action = self.policy(torch.FloatTensor(state)).numpy()

                # 添加探索噪声
                action = self.add_exploration_noise(action)

                # 执行动作
                next_state, reward, done, info = self.env.step(action)

                # 安全检查
                if not self.is_safe_state(next_state):
                    print(f"Episode {episode}, Step {step}: Unsafe state detected!")
                    done = True

                # 存储经验
                self.replay_buffer.push(state, action, reward, next_state, done)

                state = next_state
                episode_reward += reward

                if done:
                    break

            print(f"Episode {episode}: Total reward = {episode_reward:.2f}")

    def fine_tune(self, batch_size=256, num_iterations=100):
        """微调策略"""
        for iteration in range(num_iterations):
            # 从回放缓冲区采样
            batch = self.replay_buffer.sample(batch_size)
            states, actions, rewards, next_states, dones = batch

            # 计算目标Q值
            with torch.no_grad():
                next_actions = self.policy(next_states)
                target_q = self.value_network(next_states, next_actions)
                target_q = rewards + (1 - dones) * 0.99 * target_q

            # 更新Q网络
            current_q = self.value_network(states, actions)
            q_loss = F.mse_loss(current_q, target_q)

            self.value_network.optimizer.zero_grad()
            q_loss.backward()
            self.value_network.optimizer.step()

            # 策略优化
            policy_loss = -self.value_network(states, self.policy(states)).mean()

            self.policy.optimizer.zero_grad()
            policy_loss.backward()
            self.policy.optimizer.step()

            if iteration % 10 == 0:
                print(f"Iteration {iteration}, Q-Loss: {q_loss.item():.4f}, "
                      f"Policy-Loss: {policy_loss.item():.4f}")
```

### 4. Sim-to-Real 策略总结

| 方法 | 核心思想 | 适用场景 | 优点 | 缺点 |
|------|----------|----------|------|------|
| **域随机化** | 随机化仿真参数使策略泛化 | 参数不确定、视觉变化 | 简单有效 | 需要大量仿真 |
| **系统识别** | 精确建模真实物理 | 物理参数可测量 | 精度高 | 需要离线数据 |
| **烧入效应** | 保守启动策略 | 高风险任务 | 安全 | 收敛慢 |
| **在线微调** | 在真实环境调整 | 样本可用 | 最优适应 | 样本需求大 |
| **对抗方法** | 对抗训练域不变特征 | 视觉变化大 | 泛化强 | 训练复杂 |

## 代码示例

### 完整的 Sim-to-Real 训练流程

```python
class SimToRealTrainer:
    """完整的Sim-to-Real训练框架"""

    def __init__(self, sim_config, real_config):
        self.sim_env = self.create_sim_env(sim_config)
        self.real_env = self.create_real_env(real_config)

        # 网络初始化
        self.policy = ActorCritic(state_dim=20, action_dim=7)
        self.replay_buffer = ReplayBuffer(capacity=1000000)

        # 域随机化器
        self.domain_randomizer = DomainRandomizer()

        # 系统识别器
        self.sys_id = SystemIdentifier(self.sim_env.dynamics_model)

        # 微调器
        self.fine_tuner = RealWorldFineTuner(
            self.policy,
            self.policy.value_network,
            self.real_env
        )

    def phase1_sim_training(self, num_timesteps=10000000):
        """
        阶段1：大规模仿真训练
        使用域随机化训练基础策略
        """
        print("Phase 1: Simulation Training with Domain Randomization")

        state = self.sim_env.reset()

        for timestep in range(num_timesteps):
            # 随机化仿真参数（每1000步）
            if timestep % 1000 == 0:
                self.domain_randomizer.randomize_all(self.sim_env)

            # 策略选择
            action = self.policy.select_action(state)

            # 环境交互
            next_state, reward, done, _ = self.sim_env.step(action)

            # 存储
            self.replay_buffer.push(state, action, reward, next_state, done)

            state = next_state
            if done:
                state = self.sim_env.reset()

            # 策略更新
            if timestep % 2048 == 0:
                self.policy.update(self.replay_buffer)

            if timestep % 50000 == 0:
                print(f"Simulation timestep: {timestep}")

    def phase2_system_identification(self, num_trajectories=100):
        """
        阶段2：系统识别
        估计真实物理参数
        """
        print("Phase 2: System Identification")

        # 收集真实机器人数据
        print("Collecting real robot trajectories...")
        real_trajectories = self.sys_id.collect_trajectory_data(
            self.real_env,
            num_trajectories
        )

        # 估计参数
        print("Estimating physical parameters...")
        estimated_params = self.sys_id.estimate_parameters(real_trajectories)

        # 更新仿真环境
        self.sim_env.dynamics_model.set_params(estimated_params)
        print(f"Estimated parameters: {estimated_params}")

    def phase3_burn_in(self):
        """
        阶段3：烧入阶段
        保守地启动策略
        """
        print("Phase 3: Burn-in Phase")

        burn_in = BurnInStrategy(self.sim_env, self.real_env)
        burn_in_data = burn_in.collect_burn_in_data(self.policy)

        # 预填充回放缓冲区
        for s, a, r, ns in zip(
            burn_in_data['states'],
            burn_in_data['actions'],
            burn_in_data['rewards'],
            burn_in_data['next_states']
        ):
            self.replay_buffer.push(s, a, r, ns, False)

        return burn_in_data

    def phase4_real_fine_tuning(self, num_episodes=50):
        """
        阶段4：真实环境微调
        """
        print("Phase 4: Real-World Fine-tuning")

        for episode in range(num_episodes):
            # 收集真实数据
            self.fine_tuner.collect_real_data(num_episodes=1)

            # 在线微调
            self.fine_tuner.fine_tune(batch_size=256, num_iterations=50)

            # 评估
            eval_reward = self.evaluate_real_policy()
            print(f"Episode {episode}: Evaluation reward = {eval_reward:.2f}")

    def train(self):
        """完整训练流程"""
        self.phase1_sim_training()
        self.phase2_system_identification()
        self.phase3_burn_in()
        self.phase4_real_fine_tuning()

        print("Sim-to-Real training complete!")
        return self.policy
```

## 应用场景

### 1. 自动驾驶

- 仿真环境训练自动驾驶策略（CARLA、LGSVL）
- 域随机化处理不同天气、光照条件
- 真实道路测试微调

### 2. 机器人操作

- 抓取、放置、装配等操作技能学习
- 模拟学习显著减少真实机器人训练时间
- 在工业装配线上广泛应用

### 3. 人形机器人
