# 向量数据库

## 一句话定义

向量数据库是专门用于存储和检索高维向量（如文本、图像的嵌入表示）的数据库系统，通过近似最近邻(ANN)算法实现高效的相似性搜索，在AI Agent中承担记忆存储和检索的核心职能。

## 核心公式

### 向量相似度度量

#### 余弦相似度

$$
\text{cosine\_similarity}(A, B) = \frac{A \cdot B}{\|A\| \times \|B\|} = \frac{\sum_{i=1}^{n} A_i \times B_i}{\sqrt{\sum_{i=1}^{n} A_i^2} \times \sqrt{\sum_{i=1}^{n} B_i^2}}
$$

#### 欧氏距离

$$
d(A, B) = \|A - B\| = \sqrt{\sum_{i=1}^{n}(A_i - B_i)^2}
$$

#### 点积相似度

$$
\text{dot\_product}(A, B) = A \cdot B = \sum_{i=1}^{n} A_i \times B_i
$$

### ANN索引核心公式

#### HNSW分层可导航小世界图

$$
\text{Search}(q, M, efConstruction) = \text{贪婪遍历}(q, \text{layer}[L], efConstruction)
$$

$$
\text{layer}[l] = \begin{cases}
\text{完全图} & l = 0 \\
\text{稀疏连接图} & l > 0
\end{cases}
$$

#### IVF倒排文件索引

$$
\text{IVF}(q, nprobe) = \bigcup_{i=1}^{nprobe} \text{search\_cluster}(q, \text{centroid}[i])
$$

## 详细说明

### 1. 向量数据库基础

#### 1.1 什么是向量嵌入

向量嵌入是将高维数据（如文本、图像）映射到低维连续向量空间的过程：

```python
from sentence_transformers import SentenceTransformer
import numpy as np

# 加载预训练模型
model = SentenceTransformer('all-MiniLM-L6-v2')

# 将文本转换为向量
texts = ["人工智能改变世界", "AI is transforming the world"]
embeddings = model.encode(texts)

print(f"向量维度: {embeddings.shape}")  # (2, 384)
print(f"向量类型: {type(embeddings)}")  # numpy.ndarray

# 计算相似度
similarity = np.dot(embeddings[0], embeddings[1]) / (
    np.linalg.norm(embeddings[0]) * np.linalg.norm(embeddings[1])
)
print(f"余弦相似度: {similarity:.4f}")
```

### 2. 近似最近邻(ANN)算法

#### 2.1 HNSW (Hierarchical Navigable Small World)

一种基于图的高效索引算法：

```python
import faiss

# 创建HNSW索引
dimension = 384
index = faiss.IndexHNSWFlat(dimension, 32)  # 32是每个节点的连接数

# 添加向量
vectors = np.random.rand(10000, dimension).astype('float32')
index.add(vectors)

# 搜索
query = np.random.rand(1, dimension).astype('float32')
distances, indices = index.search(query, k=10)

print(f"最近邻索引: {indices[0]}")
print(f"距离: {distances[0]}")
```

#### 2.2 IVF (Inverted File Index)

基于聚类的倒排索引：

```python
# 创建IVF索引
nlist = 100  # 聚类中心数量
quantizer = faiss.IndexFlatIP(dimension)  # 用作量化器
index = faiss.IndexIVFFlat(quantizer, dimension, nlist)

# 训练
index.train(vectors)
index.add(vectors)

# 设置搜索范围
index.nprobe = 10  # 搜索的聚类数量

# 搜索
distances, indices = index.search(query, k=10)
```

#### 2.3 LSH (Locality-Sensitive Hashing)

局部敏感哈希：

```python
# 创建LSH索引
d = 384  # 向量维度
nb = 10000  # 数据库大小
n_bits = 128  # 哈希位数

# 创建随机投影矩阵
np.random.seed(1234)
random_planes = np.random.randn(d, n_bits).astype('float32')

# 手动计算LSH哈希
def lsh_hash(vector, planes):
    projections = np.dot(vector, planes)
    return (projections > 0).astype(int)

# 哈希查询向量
query_hash = lsh_hash(query[0], random_planes)
print(f"LSH哈希: {query_hash}")
```

### 4. Milvus使用示例

```python
from pymilvus import connections, Collection, FieldSchema, CollectionSchema, DataType

# 连接Milvus
connections.connect(host='localhost', port='19530')

# 定义schema
fields = [
    FieldSchema(name="id", dtype=DataType.INT64, is_primary=True),
    FieldSchema(name="embedding", dtype=DataType.FLOAT_VECTOR, dim=384),
    FieldSchema(name="text", dtype=DataType.VARCHAR, max_length=1000)
]
schema = CollectionSchema(fields=fields, description="AI Agent记忆存储")

# 创建collection
collection = Collection(name="agent_memory", schema=schema)

# 创建索引
index_params = {
    "index_type": "HNSW",
    "metric_type": "COSINE",
    "params": {"M": 16, "efConstruction": 200}
}
collection.create_index(field_name="embedding", index_params=index_params)

# 插入数据
entities = [
    [1, 2, 3],  # ids
    [[0.1]*384, [0.2]*384, [0.3]*384],  # embeddings
    ["文本1", "文本2", "文本3"]  # texts
]
collection.insert(entities)

# 搜索
search_params = {"metric_type": "COSINE", "params": {"ef": 200}}
query_vector = [[0.15]*384]
results = collection.search(query_vector, "embedding", search_params, limit=10)

for result in results[0]:
    print(f"ID: {result.id}, Distance: {result.distance}")
```

### 5. Pinecone使用示例

```python
import pinecone

# 初始化
pinecone.init(api_key="your-api-key", environment="us-west1")

# 创建索引
pinecone.create_index(
    name="agent-memory",
    dimension=384,
    metric="cosine",
    pod_type="p1"
)

# 连接索引
index = pinecone.Index("agent-memory")

# 插入向量
vectors = [
    ("doc1", [0.1]*384, {"text": "第一个文档", "category": "技术"}),
    ("doc2", [0.2]*384, {"text": "第二个文档", "category": "商业"}),
]
index.upsert(vectors)

# 搜索
query_vector = [0.15]*384
results = index.query(vector=query_vector, top_k=5, include_metadata=True)

for result in results["matches"]:
    print(f"Score: {result['score']:.4f}, Text: {result['metadata']['text']}")
```

### 6. 在Agent架构中的应用

```python
class VectorMemory:
    """Agent的向量记忆模块"""

    def __init__(self, db_type: str = "milvus"):
        if db_type == "milvus":
            self.db = MilvusDB()
        elif db_type == "pinecone":
            self.db = PineconeDB()
        self.embedding_model = SentenceTransformer('all-MiniLM-L6-v2')

    def add_memory(self, content: str, memory_type: str = "experience"):
        """添加记忆"""
        vector = self.embedding_model.encode([content])[0]
        self.db.insert(
            id=str(uuid.uuid4()),
            vector=vector,
            metadata={"content": content, "type": memory_type}
        )

    def retrieve(self, query: str, memory_type: str = None, top_k: int = 5):
        """检索相关记忆"""
        query_vector = self.embedding_model.encode([query])[0]
        results = self.db.search(query_vector, top_k=top_k)

        # 过滤类型
        if memory_type:
            results = [r for r in results if r.metadata.get("type") == memory_type]

        return results

    def retrieve_conversation_history(self, query: str, window: int = 10):
        """检索对话历史"""
        return self.retrieve(query, memory_type="conversation", top_k=window)
```
